# 처음 블로그 글 

```java
public class HelloWorld {
    public static void main(String[] args) {
        // "Hello, World!"를 출력합니다.
        System.out.println("Hello, World!");

        // 두 수를 더하는 간단한 계산
        int num1 = 5;
        int num2 = 10;
        int sum = num1 + num2;

        // 결과를 출력합니다.
        System.out.println("Sum of " + num1 + " and " + num2 + " is: " + sum);
    }
}
```
