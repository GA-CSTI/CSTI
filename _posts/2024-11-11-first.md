# 오늘 처음 블로그를 만들었습니다

```python
import sys
input = sys.stdin.readline


def sumNum (num) :
    maxNum = num[0] #최댓값
    result = num[0] #중간값

    for i in range(1, len(num)) :
        result = max(num[i], result + num[i])

        maxNum = max(maxNum, result)

    return maxNum #최댓값 반환


def main() :
    N = int(input().strip())
    num = list(map(int, input().split()))
    print(sumNum(num))

main()
```
